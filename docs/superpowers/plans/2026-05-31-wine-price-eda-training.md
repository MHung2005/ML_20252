```python
def prepare_original_frame(df: pd.DataFrame) -> pd.DataFrame:

    out["price_vnd"] = pd.to_numeric(out["price_vnd"], errors="coerce")
```

```python
def build_preprocess(numeric_features: list[str], categorical_features: list[str], text_feature: str | None = None):
    transformers = [
        ("num", Pipeline([
            ("imputer", SimpleImputer(strategy="median")),
            ("scaler", StandardScaler()),
        ]), numeric_features),
        ("cat", Pipeline([
            ("imputer", SimpleImputer(strategy="most_frequent")),
            ("onehot", OneHotEncoder(handle_unknown="ignore", min_frequency=2)),
        ]), categorical_features),
    ]
    if text_feature is not None:
        transformers.append((
            "text",
            TfidfVectorizer(max_features=40, ngram_range=(1, 2), lowercase=True),
            text_feature,
        ))
    return ColumnTransformer(transformers, remainder="drop")
```

```python
def add_engineered_features(df: pd.DataFrame) -> pd.DataFrame:
    out = df.copy()
    out["alcohol_pct"] = out["alcohol_content"] * 100

    out["volume_factor"] = out["volume"] / 750
    out["log_volume"] = np.log1p(out["volume"])
    out["is_half_bottle"] = (out["volume"] == 375).astype(int)
    out["is_standard_bottle"] = (out["volume"] == 750).astype(int)

        bins=[-1, 2, 5, 10, 20, 100],
        labels=["0-2", "3-5", "6-10", "11-20", "20+"],

    out["is_old_wine"] = (out["wine_age"] >= 10).astype(int)
```

- [ ] **Step 2: Add leakage-safe target-stat encoder**

The group statistical features from `docs/wine_feature_engineering.md` use `price_vnd`, so they must be fitted only on training folds. Add them as a scikit-learn transformer instead of precomputing them on the full dataset.

```python
class GroupTargetStatsEncoder(BaseEstimator, TransformerMixin):
    def __init__(self, columns: list[str]):
        self.columns = columns

    def fit(self, X, y):
        X_df = pd.DataFrame(X).copy()
        y_series = pd.Series(np.asarray(y), index=X_df.index, name="target")
        self.global_mean_ = np.log1p(y_series).mean()
        self.stats_ = {}
        for col in self.columns:
            tmp = pd.DataFrame({col: X_df[col].astype(str), "target": np.log1p(y_series)})
            stats = tmp.groupby(col)["target"].agg(["median", "mean", "std", "count"])
            stats = stats.rename(columns={
                "median": f"{col}_log_price_median",
                "mean": f"{col}_log_price_mean",
                "std": f"{col}_log_price_std",
                "count": f"{col}_count",
            })
            self.stats_[col] = stats
        return self

    def transform(self, X):
        X_df = pd.DataFrame(X).copy()
        output_parts = []
        for col in self.columns:
            stats = self.stats_[col]
            mapped = X_df[[col]].astype(str).join(stats, on=col).drop(columns=[col])
            for stat_col in mapped.columns:
                if stat_col.endswith("_count"):
                    mapped[stat_col] = mapped[stat_col].fillna(0)
                elif stat_col.endswith("_std"):
                    mapped[stat_col] = mapped[stat_col].fillna(0)
                else:
                    mapped[stat_col] = mapped[stat_col].fillna(self.global_mean_)
            output_parts.append(mapped.reset_index(drop=True))
        return pd.concat(output_parts, axis=1)
```

```python
def evaluate_predictions(y_true, y_pred) -> dict:
```

```python
def evaluate_feature_set(feature_set_name, models, X_train, X_test, y_train, y_test):
    rows = []
    fitted = {}
    for model_name, model in models.items():
        scores = cross_validate(model, X_train, y_train, cv=cv, scoring=scoring, n_jobs=-1)
        model.fit(X_train, y_train)
        fitted[model_name] = model
        pred = model.predict(X_test)
        metrics = evaluate_predictions(y_test, pred)
        rows.append({
            "FeatureSet": feature_set_name,
            "Model": model_name,
            "CV_MAE_mean": -scores["test_MAE"].mean(),
            "CV_MAE_std": scores["test_MAE"].std(),
            "CV_RMSE_mean": -scores["test_RMSE"].mean(),
            "CV_R2_mean": scores["test_R2"].mean(),
            "Holdout_MAE": metrics["MAE"],
            "Holdout_RMSE": metrics["RMSE"],
            "Holdout_R2": metrics["R2"],
            "Holdout_MAPE": metrics["MAPE"],
        })
    result = pd.DataFrame(rows).sort_values("Holdout_MAE")
    return result, fitted
```

```python
def plot_prediction_diagnostics(model, X_test, y_test, title, output_name):
    pred = model.predict(X_test)
    residuals = y_test - pred

    fig, axes = plt.subplots(1, 2, figsize=(13, 5))
    axes[0].scatter(y_test / 1e6, pred / 1e6, alpha=0.75)
    mn = min(y_test.min(), pred.min()) / 1e6
    mx = max(y_test.max(), pred.max()) / 1e6
    axes[0].plot([mn, mx], [mn, mx], "r--")
```
